def bgr_2_rgb(img):
    
    """ Rcupre les valeurs BGR et switch  RGB.
    
    Args:
        X:  Image OpenCV (BGR) de la matrice X. (ex: X[83])

def center_crop(img, new_h=None, new_w=None):        
    
    """ Fonction qui recadre une image en son centre.
        Supporte aussi les images non carr.
    
    Args:
        img:   OpenCV Image RGB 
        new_h: La nouvelle hauteur de l'image
        new_w: La nouvelle largeur de l'image  
    Returns:
        Image recadre en son centre avec la nouvelle hauteur et largeur
    """   
    h = img.shape[0]
    w = img.shape[1]
    if new_w is None: new_w = min(w,h)
    if new_h is None: new_h = min(w,h)

    l = int(np.ceil((w - new_w) / 2))      # Left
    r = w - int(np.floor((w - new_w) / 2)) # Right
    t = int(np.ceil((h - new_h) / 2))      # Top
    b = h - int(np.floor((h - new_h) / 2)) # Bottom

    if len(img.shape) == 2: img_crop = img[t:b,l:r]
    else: img_crop = img[t:b,l:r,...]
        
    return img_crop


"""""""""""""""""""""
"  IMG CIRCLE CROP  "
"""""""""""""""""""""
def circle_crop(img,r):

    """ Fonction qui applique un crop de forme circulaire au centre de l'image.
        On peinture en noir les pixels  l'extrieur du cercle de l'image.
    
    Args:
        img: OpenCV Image RGB
        r:   Rayon du cercle (crop)
    Returns:
        Image avec un crop de forme circulaire 
    """  
    img_circle = img.copy()
    c = (img.shape[0]/2) - r

    for i in range(img.shape[0]):
        for j in range(img.shape[1]):
            
            c_y = math.pow(i-c-r,2)
            c_x = math.pow(j-c-r,2)
            d = math.sqrt(c_x + c_y)           
            if(d > r): img_circle[i][j] = [0,0,0]
                
    return img_circle

def averaging_filter(img, kernel):
    
    """ Fonction qui applique un filtre moyen sur les valeurs 
        des pixels de l'image dans une fentre (matrice) donne.  
    
    Args:
        img:       OpenCV Image RGB
        kernel:    La taille de la matrice (kernel --> ex: 5 = matrice 5 x 5)   
    Returns:
        Image avec une moyenne applique sur la valeur de ses pixels
    """    
    k = np.ones((kernel, kernel), np.float32) / (kernel * kernel)
    img_average = cv2.filter2D(img, -1, k)
    return img_average

def gaussian_filter(img, mat_size, border_type):
    
    """ Fonction qui applique un filtre Gaussien sur les valeurs 
        des pixels de l'image dans une fentre (matrice) donne et
        avec un type de bordure donne.  
    
    Args:
        img:         OpenCV Image RGB
        mat_size:    La taille de la matrice. (ex: 5 = matrice 5 x 5)
        border_type: Le type bordure  appliquer (remplir).      
    Returns:
        Image avec un filtre Gaussien applique averc une bordure donne
    """   
    gaussian_img = cv2.GaussianBlur(img, (mat_size, mat_size), border_type)
    return gaussian_img

En effet, des primitives ont t choisies pour chacun des trois aspects morphologiques prsents plus tt. Ainsi, lexcentricit et la circularit sont les primitives choisies pour reprsenter la forme. La primitive de lexcentricit est propose par Gauthier, Jain et Noordeh (2016). Lexcentricit exprime si une forme est plus un cercle ou une ellipse. Lexcentricit est dfinie comme :
$$e = \sqrt{1 - \frac{b^{2}}{a^{2}}}$$ o un cercle une excentricit de 0 et une ellipse une excentricit entre 0 et 1 exclusivement.

Paralllement, Dhami (2015) propose dutiliser le rapport de la hauteur sur la largeur. Ce dernier rapport pourrait tre une primitive qui caractrise la forme de la galaxie. Cependant, ce rapport ne prend pas en compte la position des foyers de lellipse comme le fait lexcentricit. Puisque l'excentricit inclut dans sa dfinition ce mme rapport, utiliser ce rapport ainsi que l'excentricit comme primitive semble redondant. En ce qui concerne la circularit, la dfinition choisie pour cette primitive est celle de uni et Hirota (2008). Cette dfinition est une amlioration face une autre dfinition bien connue. Pour viter toute confusion, cette autre dfinition de la circularit sera rfre par compacit, dans le reste du rapport. De ce fait, selon uni et Hirota : Probably the most standard circularity measure (sometimes also called the compactness) is derived from the relation between the shape area and its perimeter (uni et Hirota, 2008, p.94). Ainsi, la compacit est dfinie par l'quation :
$$C = 4*\pi * \frac{A}{P^{2}}$$ o A est laire de la forme et P son primtre.

La compacit est entre autres prsente par Dhami (2015) comme une primitive pour la circularit. Pour leur part, uni et Hirota font usage de la notion de probabilit des moments centrs :
$$C(S) = \frac{1}{2\pi} * \frac{(\mu_{0,0}(S))^2}{\mu_{2,0}(S)+\mu_{0,2}(S)}$$ $$\mu_{p,q}=\sum_{x}\sum_{y}(x-\frac{M_{10}}{M_{00}})^{p}\thinspace(y-\frac{M_{01}}{M_{00}})^{q}$$ $$M_{i,j}=\sum_{x}\sum_{y}x^{i}y^{j}I(x,y)$$ o $I(x,y)$ est lintensit des pixels dune image en nuances de gris.

La centralit donne une mesure entre [0, 1] o 1 est un cercle parfait. Cette dfinition de la circularit a t choisie face la compacit puisquelle est robuste au bruit. De plus, la compacit [traduit] pnalise les intrusions profondes dans la forme. De telles intrusions et protrusion mnent une augmentation du primtre, et en consquence, impliquent une mesure de [compacit] plus basse (uni et Hirota, 2008, p.98). La circularit n'a pas ce problme. Enfin, la circularit [traduit] est invariante avec respect aux translations, rotations, et changement dchelle (uni et Hirota, 2008, p.100).

Ensuite, la suggestion de Gauthier, Jain et Noordeh (2016) pour la primitive de la luminosit a t choisie. Ainsi, cette primitive est la somme des valeurs des pixels de limage. def get_image_red_to_blue_ratio(rgb_img) :
    
    """ Fonction qui retourne le ratio red to blue d'une image
    
    Args:
        rgb_img: L'image en format RGB pour laquelle le ratio red to blue est calcul       
    Returns:

def get_img_eccentricity(bgr_img, show_image) :
    
    """ Fonction qui retourne l'excentricit d'une galaxy
    
    Args:
        bgr_img:    Image en format BGR pour laquelle l'excentricit est calcul
        show_image: Show debug image        
    Returns:
        L'excentricit d'une galaxy
    """
    #Transforme les images en niveau de gris
    new_image = bgr_img.astype(np.uint8)
    gray = cv.cvtColor(new_image, cv2.COLOR_BGR2GRAY)
    
    #Cre une image binaire
    ret,th = cv2.threshold(gray,50,255, 0)
    #ret,th = cv2.threshold(gray,0,255,cv2.THRESH_BINARY+cv2.THRESH_OTSU)
    
    #Trouve les contours dans cette image binaire
    _, contours,hierarchy = cv2.findContours(th,2,1)
    cnt = contours
    
    #affiche l'original et les contours si show_image est vrai
    if show_image :
        f, axarr = plt.subplots(1,2, figsize=(10, 10))
        axarr[0].set_title("Original")
        axarr[1].set_title("Contour")
        axarr[0].imshow(bgr_img / 255)
        axarr[1].imshow(cv2.drawContours(new_image, cnt, -1, (0,255,0), 3) / 255)
        plt.show()
    
    #Trouve le contour qui produit l'ellipse la plus proche du centre
    center_ellipse = []
    min_distance_from_center = 99999
    for i in cnt :
        #On ne peut pas cre d'elipse si le contour fait moins de 5 point
        if i.shape[0] < 5 :
            continue
        ellipse = cv2.fitEllipse(i)
        ellipse_x = ellipse[0][0]
        ellipse_y = ellipse[0][1]
        image_x_center = new_image.shape[0] / 2
        image_y_center = new_image.shape[1] / 2
        distance_from_center = math.sqrt((image_x_center - ellipse_x)**2 + (image_y_center - ellipse_y)**2)
        if distance_from_center < min_distance_from_center :
            min_distance_from_center = distance_from_center
            center_ellipse = ellipse
    
    #Affiche l'image des contours et celle de l'elipse choisi si show_image est vrai
    if show_image :
        ellipse_image = bgr_img.copy()
        cv2.ellipse(ellipse_image, center_ellipse, (255,0,0), 2)
        f, axarr = plt.subplots(1,2, figsize=(10, 10))
        axarr[0].set_title("Contour")
        axarr[1].set_title("Ellipse")
        axarr[0].imshow(cv2.drawContours(new_image, cnt, -1, (0,255,0), 3) / 255)
        axarr[1].imshow(cv2.ellipse(ellipse_image, center_ellipse, (255,0,0), 2) / 255)
        plt.show()
    
    #calcul l'excentricit
    (x, y), (MA, ma), angle = center_ellipse
    a = ma/2
    b = MA/2
    eccentricity = math.sqrt(1- (pow(b, 2)/pow(a, 2)))
    
    return eccentricity

def get_img_circularity(bgr_img):

    Calcule la circularit de l'image selon la dfinition de la circularit de Zunic et Hirota dans
    
    uni J., Hirota K. (2008) Measuring Shape Circularity. In: Ruiz-Shulcloper J., Kropatsch W.G. (eds)
    Progress in Pattern Recognition, Image Analysis and Applications. CIARP 2008. Lecture Notes in 
    Computer Science, vol 5197. Springer, Berlin, Heidelberg
    
    Args:
        bgr_img: Image en format BGR pour laquelle on dsire calculer la circularit.       
    Returns:
        La circularit de l'image
    """
    
    gray_img = cv.cvtColor(bgr_img, cv2.COLOR_BGR2GRAY)
    ret, th = cv2.threshold(gray_img,0,255,cv2.THRESH_BINARY+cv2.THRESH_OTSU)
    moments = cv2.moments(th, binaryImage=True)
    circularity = (1.0 / (2.0*math.pi)) * (math.pow(moments['m00'], 2) / (moments['mu20'] + moments['mu02']))
    
    return circularity

def train(feature_vector, y_label_encoded, max_depths, nb_folds):
    
    """
    Entraine l'arbre de dcision et effectue une validation croise
    de type stratified k fold pour chaque profondeur maximale spcifie.
    
    Args:

        max_depths:      Une liste des profondeurs maximales de l'arbre.
        nb_folds:        Le nombre de folds  faire lors de la validation.    
    
    Returns:

    """

    model = tree.DecisionTreeClassifier()

def min_max_normalize(data):
    
    """
    Effectuez une normalisation min-max, qui normalise les donnes entre 0 et 1, 
    sur l'ensemble de donnes donn.
    
    Args:
        data: The dataset to normalize      
    Returns:
        The normalized dataset
    """
    data_min = min(data)
    data_max = max(data)
    diff = data_max-data_min
    return [(x-data_min)/(diff) for x in data]
